package io.blazornative.shell

import android.app.Activity
import android.os.Bundle
import android.util.Log
import android.widget.TextView
import io.blazornative.jni.WasiHost
import kotlin.concurrent.thread

class MainActivity : Activity() {

    private val log = "BlazorNative"

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.main)

        val view = findViewById<TextView>(R.id.markers)

        // .wasm boot can take ~500ms-1500ms (Cranelift JIT cold compile + Mono-AOT init)
        // — keep it off the UI thread.
        thread(name = "BlazorNative-WasiHost-Boot") {
            try {
                val wasmBytes = assets.open("BlazorNative.WasiHost.wasm").use { it.readBytes() }
                Log.i(log, "Loaded ${wasmBytes.size} bytes of .wasm from assets; booting...")

                val stdout = WasiHost.loadAndRun(wasmBytes, cacheDir)

                // Each captured line → one Log.i call (atomic per-line in logcat)
                stdout.lineSequence().filter { it.isNotBlank() }.forEach { line ->
                    Log.i(log, line)
                }

                runOnUiThread { view.text = stdout }
            } catch (t: Throwable) {
                Log.e(log, "Boot failed", t)
                runOnUiThread { view.text = "FAIL: ${t.javaClass.simpleName}: ${t.message}" }
            }
        }
    }
}

4. **Task 9 (connectedAndroidTest — THE GREEN CHECKPOINT, most iteration-prone).** Six potential failure modes documented inline. ~60 min iteration budget; if exhausted, fallback is leaner test ("no-trap") OR direct wasmtime CLI on emulator to isolate JNA vs runtime issues.
